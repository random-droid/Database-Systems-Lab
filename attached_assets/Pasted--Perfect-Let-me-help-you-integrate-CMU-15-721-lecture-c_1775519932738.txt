## **UI Components to Build:**

### **1. LectureContextBox Component:**

```jsx
const LectureContextBox = ({ lecture, useCase }) => {
  const [expanded, setExpanded] = useState(false);
  
  return (
    <div className="lecture-context-box">
      <div className="lecture-header" onClick={() => setExpanded(!expanded)}>
        <span className="lecture-number">
          📚 Lecture {lecture.number}
        </span>
        <span className="lecture-title">
          {lecture.title}
        </span>
        <button className="toggle-btn">
          {expanded ? '▼' : '▶'}
        </button>
      </div>
      
      {expanded && (
        <div className="lecture-details">
          <blockquote className="key-quote">
            "{lecture.quote}"
          </blockquote>
          
          <div className="concepts">
            <h4>Key Concepts:</h4>
            <ul>
              {lecture.key_concepts.map(concept => (
                <li key={concept}>{concept}</li>
              ))}
            </ul>
          </div>
          
          <div className="summary">
            {lecture.summary}
          </div>
          
          {lecture.smoking_gun_context && (
            <div className="context-tip">
              💡 <strong>What to look for:</strong> {lecture.smoking_gun_context}
            </div>
          )}
          
          <a href={lecture.url} target="_blank" className="lecture-link">
            📺 Watch Lecture {lecture.number}
          </a>
        </div>
      )}
    </div>
  );
};
```
