# Fallback sizes
DATASET_SIZES = {
    'full': 50_000_000,      # Target
    'medium': 10_000_000,    # If memory tight
    'small': 1_000_000       # Minimum viable
}

def auto_select_dataset_size():
    """Auto-detect safe dataset size"""
    mem = psutil.virtual_memory()
    
    if mem.total < 2.5 * 1024**3:  # < 2.5GB
        return DATASET_SIZES['small']
    elif mem.total < 4 * 1024**3:  # < 4GB
        return DATASET_SIZES['medium']
    else:
        return DATASET_SIZES['full']

            'version': spark.version

        status['spark'] = {

            'error': str(e),
            'suggestion': 'Check: pip install pyspark delta-spark'

    return status

## **Quick Validation Script:**

```python
# test_pipeline.py - Run before full benchmarks
import psutil
import subprocess

def validate_environment():
    """Pre-flight check"""
    
    issues = []
    
    # 1. Memory check
    mem = psutil.virtual_memory()
    if mem.total < 2 * 1024**3:
        issues.append(f"⚠️  Low memory: {mem.total / 1024**3:.1f}GB")
    
    # 2. Disk space
    disk = psutil.disk_usage('/')
    if disk.free < 10 * 1024**3:
        issues.append(f"⚠️  Low disk: {disk.free / 1024**3:.1f}GB free")
    
    # 3. Postgres availability
    try:
        subprocess.run(['psql', '--version'], check=True, capture_output=True)
    except:
        issues.append("❌ Postgres not available")
    
    # 4. Python packages
    required = ['pandas', 'pyarrow', 'duckdb', 'pyspark']
    for pkg in required:
        try:
            __import__(pkg)
        except:
            issues.append(f"❌ Missing: {pkg}")
    
    if issues:
        print("🚨 Pre-flight check failed:")
        for issue in issues:
            print(f"   {issue}")
        return False
    else:
        print("✅ Environment ready")
        return True

if __name__ == '__main__':
    validate_environment()
```
